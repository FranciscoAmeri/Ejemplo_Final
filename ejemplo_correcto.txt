
// Variables globales
int contadorGlobal;
double valorPi;
char inicial;
bool activo;

// Función simple
int sumar(int a, int b) {
    int resultado;
    resultado = a + b;
    contadorGlobal = contadorGlobal + 1;
    return resultado;
}

// Función con arrays
void procesarArray(int datos[10]) {
    int i;
    int suma;
    
    suma = 0;
    for (i = 0; i < 10; i++) {
        suma = suma + datos[i];
    }
    
    valorPi = suma;
}

// Función con punteros
void trabajarConPunteros() {
    int valor;
    int* puntero;
    
    valor = 42;
    puntero = valor;
    contadorGlobal = puntero;
}

// Función con diferentes tipos
void manejarTipos() {
    int entero;
    double decimal;
    char caracter;
    bool bandera;
    float flotante;
    
    entero = 100;
    decimal = 3.14159;
    caracter = 'A';
    bandera = true;
    flotante = 2.5;
    
    // Usar todas las variables
    valorPi = decimal + flotante;
    inicial = caracter;
    activo = bandera;
    contadorGlobal = entero;
}

// Función con estructuras de control
void estructurasControl() {
    int contador;
    bool continuar;
    
    contador = 0;
    continuar = true;
    
    // Bloque if
    if (continuar) {
        int temp;
        temp = contador + 5;
        contador = temp;
    }
    
    // Bucle for
    for (int i = 0; i < 5; i++) {
        contador = contador + i;
    }
    
    // Bucle while
    while (contador > 0) {
        contador = contador - 1;
        if (contador == 10) {
            int auxiliar;
            auxiliar = contador * 2;
            contador = auxiliar;
        }
    }
    
    contadorGlobal = contador;
}

// Función que llama a otras funciones
void coordinador() {
    int resultado;
    int numeros[10];
    int indice;
    
    // Inicializar array
    for (indice = 0; indice < 10; indice++) {
        numeros[indice] = indice * 2;
    }
    
    // Llamar funciones
    resultado = sumar(10, 20);
    procesarArray(numeros);
    trabajarConPunteros();
    manejarTipos();
    estructurasControl();
    
    // Usar el resultado
    contadorGlobal = resultado;
}

// Función main
int main() {
    int estado;
    
    // Inicializar variables globales
    contadorGlobal = 0;
    valorPi = 3.14;
    inicial = 'M';
    activo = false;
    
    // Ejecutar el programa
    coordinador();
    
    estado = contadorGlobal;
    return estado;
}
